193 template <typename Complex, std::size_t BaseLevel>
194 struct InnerVisitor
195 {
196  using SimplexSet = typename casc::SimplexSet<Complex>;
197  using SimplexMap = typename casc::SimplexMap<Complex>;
198  using Simplex = typename Complex::template SimplexID<BaseLevel>;
199  using KeyType = typename Complex::KeyType;
200 
201  InnerVisitor(SimplexSet* p, Simplex s, KeyType np, SimplexMap* rv)
202  : pLevels(p), simplex(s), new_point(np), data(rv) {}
203 
214  template <std::size_t OldLevel>
215  bool visit(Complex &F, typename Complex::template SimplexID<OldLevel> s)
216  {
217  constexpr std::size_t NewLevel = OldLevel - BaseLevel + 1;
218 
219  if (pLevels->find(s) != pLevels->template end<OldLevel>())
220  {
221  //std::cout << "InnerVisitor (found): " << s << std::endl;
222  auto old_name = F.get_name(s);
223  auto base_name = F.get_name(simplex);
224  using NewArrayType = std::array<KeyType, NewLevel>;
225  NewArrayType new_name;
226 
227  std::size_t i = 0; // new_name
228  std::size_t j = 0; // old_name
229  std::size_t k = 0; // base_name
230 
231  new_name[i++] = new_point;
232 
233  // Remove base_name from old_name and append to new_name
234  while (i < NewLevel)
235  {
236  if (base_name[k] == old_name[j])
237  {
238  // if equivalent than skip the value
239  ++j; ++k;
240  }
241  else
242  {
243  // append to new_name and increment
244  new_name[i++] = old_name[j++];
245  }
246  }
247 
248  SimplexSet grab;
249  visit_BFS_down(GrabVisitor<Complex>(pLevels, &grab), F, s);
250 
251  auto &levelMap = casc::get<NewLevel>(*data);
252  auto it = levelMap.find(new_name);
253  if (it != levelMap.end())
254  {
255  it->second.insert(grab);
256  }
257  else
258  {
259  MAYBE_UNUSED auto ret = levelMap.insert(
260  std::pair<NewArrayType, SimplexSet>(new_name, grab));
261  assert(ret.second);
262  }
263  }
264  return true;
265  }
266 
267  private:
268  SimplexSet* pLevels;
269  Simplex simplex;
270  KeyType new_point;
271  SimplexMap* data;
272 };

480 template <typename Complex, typename Simplex, template<typename> class Callback>
481 void decimate(Complex &F, Simplex s, Callback<Complex> &&clbk)
482 {
484  using SimplexSet = typename casc::SimplexSet<Complex>;
486  using SimplexMap = typename casc::SimplexMap<Complex>;
487 
488  // Create the vertex to replace `s`
489  typename Complex::KeyType np = F.add_vertex();
490  SimplexSet nbhd;
491  SimplexMap simplexMap;
492 
493  // Get the complete neighborhood
494  visit_BFS_down(
495  decimation_detail::GetCompleteNeighborhood<Complex>(&nbhd),
496  F, s);
497 
498  SimplexSet doomed = nbhd; // Backup the neighborhood
499  // Call MainVisitor -> InnerVisitor -> GrabVisitor sequence
500  visit_BFS_down(
501  decimation_detail::MainVisitor<Complex>(
502  &nbhd, np, &simplexMap),
503  F, s);
504  // Run the user specified callback
505  typename decimation_detail::SimplexDataSet<Complex>::type rv;
506  run_user_callback(F, simplexMap, std::forward<Callback<Complex> >(clbk), rv);
507  perform_removal(F, doomed); // Remove simplices in the neighborhood
508  perform_insertion(F, rv); // Insert new simplices
509 }
